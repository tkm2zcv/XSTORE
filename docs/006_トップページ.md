### app/(public)/page.tsx
```typescript
import { Metadata } from 'next';
import { HeroSection } from '@/components/home/HeroSection';
import { FeaturedAccounts } from '@/components/home/FeaturedAccounts';
import { Features } from '@/components/home/Features';
import { Categories } from '@/components/home/Categories';
import { createServerClient } from '@/lib/supabase';

export const metadata: Metadata = {
  title: 'Twitter Account Market - Twitterアカウント買取・販売',
  description:
    'Twitterアカウントの買取・販売を安心・安全に行えるプラットフォーム。豊富な取扱アカウントと迅速なサポートで、あなたのビジネスを支援します。',
  keywords: 'Twitter, アカウント, 買取, 販売, フォロワー',
  openGraph: {
    title: 'Twitter Account Market - Twitterアカウント買取・販売',
    description:
      'Twitterアカウントの買取・販売を安心・安全に行えるプラットフォーム',
    type: 'website',
  },
};

async function getFeaturedAccounts() {
  const supabase = createServerClient();
  const { data, error } = await supabase
    .from('accounts')
    .select('*')
    .eq('status', 'available')
    .order('followers_count', { ascending: false })
    .limit(6);

  if (error) {
    console.error('Error fetching accounts:', error);
    return [];
  }

  return data;
}

export default async function HomePage() {
  const featuredAccounts = await getFeaturedAccounts();

  return (
    <div className="flex flex-col gap-16 py-8">
      <HeroSection />
      <FeaturedAccounts accounts={featuredAccounts} />
      <Features />
      <Categories />
    </div>
  );
}
```

### components/accounts/AccountCard.tsx（基本版）
```typescript
import Image from 'next/image';
import Link from 'next/link';
import { Card, CardContent, CardFooter } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Account } from '@/types';
import { Users, MessageSquare } from 'lucide-react';

interface AccountCardProps {
  account: Account;
}

export function AccountCard({ account }: AccountCardProps) {
  return (
    <Link href={`/accounts/${account.id}`}>
      <Card className="overflow-hidden transition-all hover:border-primary">
        <CardContent className="p-6">
          <div className="mb-4 flex items-center gap-4">
            <div className="relative h-16 w-16 overflow-hidden rounded-full">
              <Image
                src={account.image_url || '/placeholder-avatar.png'}
                alt={account.username}
                fill
                className="object-cover"
              />
            </div>
            <div className="flex-1">
              <h3 className="font-semibold">{account.username}</h3>
              <Badge variant="secondary">{account.category}</Badge>
            </div>
          </div>

          <div className="flex gap-4 text-sm text-muted-foreground">
            <div className="flex items-center gap-1">
              <Users className="h-4 w-4" />
              {account.followers_count.toLocaleString()}
            </div>
            <div className="flex items-center gap-1">
              <MessageSquare className="h-4 w-4" />
              {account.tweets_count.toLocaleString()}
            </div>
          </div>
        </CardContent>

        <CardFooter className="bg-muted/40 p-6">
          <div className="text-2xl font-bold text-primary">
            ¥{account.price.toLocaleString()}
          </div>
        </CardFooter>
      </Card>
    </Link>
  );
}
```
